# Plan 26-01 – Admin Dashboard: Basis & Gebruikersbeheer

**Phase goal:** Create the `/admin` section with service-role Supabase client, admin-only layout guard, dashboard homepage, and full user management (list, role change, delete).

**Status:** PENDING

---

## Context

All 25 MVP phases are complete (commit `b59628e`, tag `v1.0.0-mvp`). The platform has no admin interface at all. Changing user roles currently requires direct Supabase Dashboard access. This plan creates the entire admin foundation.

**Critical security constraint:** The service-role Supabase key bypasses all RLS. The `lib/supabase/admin.ts` module must NEVER be imported from client-side code. It must only be used in Server Actions and server-only functions.

**Existing patterns to follow:**
- Admin role check pattern: `lib/supabase/server.ts` createClient → `profiles` table → check `role === 'admin'`
- Server Action pattern: `app/qa/actions.ts` (authenticate → checkRole → validate → execute → return `{ error }`)
- Layout guard pattern: `app/profiel/page.tsx` (check user, redirect if not authenticated)

### Task 3 – Rolbeheer & verwijder Server Actions

**`app/admin/gebruikers/actions.ts`**

```typescript
'use server'

import { redirect } from 'next/navigation'
import { createClient } from '@/lib/supabase/server'
import { supabaseAdmin } from '@/lib/supabase/admin'

async function verifyAdmin() {
  const supabase = await createClient()
  const { data: { user } } = await supabase.auth.getUser()
  if (!user) redirect('/login')

  const { data: profile } = await supabase
    .from('profiles')
    .select('role')
    .eq('id', user.id)
    .single()

  if (profile?.role !== 'admin') redirect('/')
  return user
}

export async function updateUserRole(userId: string, role: 'user' | 'moderator' | 'admin') {
  await verifyAdmin()

  // UUID validation
  if (!/^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(userId)) {
    return { error: 'Ongeldig gebruikers-ID' }
  }

  const validRoles = ['user', 'moderator', 'admin'] as const
  if (!validRoles.includes(role)) {
    return { error: 'Ongeldige rol' }
  }

  const { error } = await supabaseAdmin
    .from('profiles')
    .update({ role, updated_at: new Date().toISOString() })
    .eq('id', userId)

  if (error) return { error: 'Kon rol niet bijwerken' }
  return { success: true }
}

export async function deleteUser(userId: string) {
  await verifyAdmin()

  if (!/^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i.test(userId)) {
    return { error: 'Ongeldig gebruikers-ID' }
  }

  const { error } = await supabaseAdmin.auth.admin.deleteUser(userId)
  if (error) return { error: 'Kon gebruiker niet verwijderen' }

  redirect('/admin/gebruikers')
}
```
