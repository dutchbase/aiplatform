Create `components/ui/breadcrumbs.tsx` as a Server Component (no 'use client' directive).

The component accepts a single prop: `items: Array<{ label: string; href?: string }>`.
- Items WITH href render as `<Link>` anchors (all items except the last).
- The LAST item (no href) renders as a plain `<span>` with `aria-current="page"`.
- The separator character `›` is wrapped in `<span aria-hidden="true">` to suppress screen reader announcement.
- The outer element is `<nav aria-label="Breadcrumb">` containing `<ol>`.
- Use ONLY semantic token Tailwind classes (decision 02-01-D4): text-muted-foreground, text-foreground, hover:text-foreground.

Exact implementation:

```tsx
import Link from 'next/link'

export interface BreadcrumbItem {
  label: string
  href?: string
}

interface BreadcrumbsProps {
  items: BreadcrumbItem[]
}

export function Breadcrumbs({ items }: BreadcrumbsProps) {
  return (
    <nav aria-label="Breadcrumb" className="mb-6">
      <ol className="flex flex-wrap items-center gap-1.5 text-sm text-muted-foreground">
        {items.map((item, index) => {
          const isLast = index === items.length - 1
          return (
            <li key={index} className="flex items-center gap-1.5">
              {index > 0 && (
                <span aria-hidden="true" className="select-none">
                  ›
                </span>
              )}
              {isLast || !item.href ? (
                <span
                  className="font-medium text-foreground"
                  aria-current={isLast ? 'page' : undefined}
                >
                  {item.label}
                </span>
              ) : (
                <Link
                  href={item.href}
                  className="hover:text-foreground transition-colors"
                >
                  {item.label}
                </Link>
              )}
            </li>
          )
        })}
      </ol>
    </nav>
  )
}
```

Note: Export both `Breadcrumbs` (the component) and `BreadcrumbItem` (the type). The component has no default export — only named exports.
